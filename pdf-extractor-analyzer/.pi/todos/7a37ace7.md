## Code Design
```python
class PDFExtractor:
    async def extract_async(self, pdf_path, **kwargs) -> ExtractionResult:
        pages = await self._convert_pages_async(pdf_path)
        semaphore = asyncio.Semaphore(self.config.max_concurrent_pages)
        
        async def process_page(page):
            async with semaphore:
                return await self.analyzer.analyze_page_async(page)
        
        results = await asyncio.gather(*[process_page(p) for p in pages])
        return self._combine_results(results)
```
